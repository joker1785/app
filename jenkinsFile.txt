pipeline {
    agent any

    stages {
        stage('Get source code') {
            steps {
                // Fetch the source code from the repository
                git 'https://github.com/joker1785/adminPortal.git'
            }
        }

        stage('Build') {
            steps {
                // Run Maven build
                bat 'mvn clean compile'
            }
        }

        stage('Test') {
            steps {
                // Run Maven tests
                bat 'mvn test'
            }
        }

        stage('Report') {
            steps {
                // Generate test report
                publishHTML(target: [
                    allowMissing: false,
                    alwaysLinkToLastBuild: true,
                    keepAll: true,
                    reportDir: 'target/surefire-reports',
                    reportFiles: 'index.html',
                    reportName: 'Test Report'
                ])
            }
        }
    }

    post {
        always {
            // Clean up the workspace after the pipeline finishes
            cleanWs()
        }
        success {
            // Print a message if the pipeline completes successfully
            echo 'Pipeline completed successfully!'
        }
        failure {
            // Print a message if the pipeline fails
            echo 'Pipeline failed!'
        }
    }
}
